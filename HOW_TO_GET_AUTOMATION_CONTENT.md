## 1. RSS Feed Items

### Via API Routes

#### Get All RSS Feeds
```typescript
// GET /api/rss/feeds
const response = await fetch('/api/rss/feeds');
const { feeds } = await response.json();
```

#### Get Feed Items for a Specific Feed
```typescript
// GET /api/rss/feeds/:id/items?status=pending
const feedId = 'your-feed-id';
const status = 'pending'; // optional: 'pending', 'processed', 'article_generated', 'skipped', 'error'

const response = await fetch(`/api/rss/feeds/${feedId}/items?status=${status}`);
const { items } = await response.json();
```

#### Get Import Jobs (History)
```typescript
// GET /api/rss/feeds/:id/jobs
const feedId = 'your-feed-id';
const response = await fetch(`/api/rss/feeds/${feedId}/jobs`);
const { jobs } = await response.json();
```

### Via Service Layer

```typescript
import { rssImportService } from '@/lib/rss-import/RSSImportService';

// Get all feeds
const feeds = await rssImportService.getFeeds();

// Get feed items
const items = await rssImportService.getFeedItems(feedId, 'pending'); // status is optional

// Get import jobs
const jobs = await rssImportService.getImportJobs(feedId);
```

### Direct Database Query (Supabase)

```typescript
import { createClient } from '@/lib/supabase/client';

const supabase = createClient();

// Get all RSS feeds
const { data: feeds } = await supabase
  .from('rss_feeds')
  .select('*')
  .order('created_at', { ascending: false });

// Get feed items with status filter
const { data: items } = await supabase
  .from('rss_feed_items')
  .select('*')
  .eq('feed_id', feedId)
  .eq('processing_status', 'pending') // optional filter
  .order('published_date', { ascending: false });

// Get items that have generated articles
const { data: itemsWithArticles } = await supabase
  .from('rss_feed_items')
  .select(`
    *,
    articles:generated_article_id (
      id,
      title,
      slug,
      status
    )
  `)
  .not('generated_article_id', 'is', null);
```

## 2. Generated Articles

### Get Articles Generated from RSS

```typescript
import { createClient } from '@/lib/supabase/client';

const supabase = createClient();

// Get articles that were generated from RSS items
const { data: articles } = await supabase
  .from('articles')
  .select(`
    *,
    rss_source:rss_feed_items!generated_article_id (
      id,
      title,
      original_url,
      feed_id,
      rss_feeds (
        name,
        url
      )
    )
  `)
  .not('rss_feed_items.generated_article_id', 'is', null)
  .eq('ai_generated', true); // if you want only AI-generated articles
```

### Get All AI-Generated Articles

```typescript
const { data: aiArticles } = await supabase
  .from('articles')
  .select('*')
  .eq('ai_generated', true)
  .order('created_at', { ascending: false });
```

### Via API (if you have an articles API)

```typescript
// Assuming you have an articles API endpoint
const response = await fetch('/api/articles?ai_generated=true');
const { articles } = await response.json();
```

## 3. Keyword Research Data

### Get Keywords for an Article

```typescript
import { createClient } from '@/lib/supabase/client';

const supabase = createClient();

// Get keyword research for a specific article
const { data: keywords } = await supabase
  .from('keyword_research')
  .select('*')
  .eq('article_id', articleId)
  .order('created_at', { ascending: false });

// Get primary keywords only
const { data: primaryKeywords } = await supabase
  .from('keyword_research')
  .select('*')
  .eq('article_id', articleId)
  .eq('keyword_type', 'primary');

// Get long-tail keywords
const { data: longTailKeywords } = await supabase
  .from('keyword_research')
  .select('*')
  .eq('article_id', articleId)
  .eq('keyword_type', 'long-tail');
```

### Get Keyword Clusters

```typescript
// Get all keyword clusters
const { data: clusters } = await supabase
  .from('keyword_clusters')
  .select(`
    *,
    articles:article_keyword_clusters (
      article_id,
      articles (
        id,
        title,
        slug
      )
    )
  `)
  .order('authority_score', { ascending: false });
```

### Get Title Variations

```typescript
// Get title variations for an article
const { data: titleVariations } = await supabase
  .from('title_variations')
  .select('*')
  .eq('article_id', articleId)
  .order('seo_score', { ascending: false });

// Get selected title
const { data: selectedTitle } = await supabase
  .from('title_variations')
  .select('*')
  .eq('article_id', articleId)
  .eq('is_selected', true)
  .single();
```

### Get Keyword Extractions from RSS

```typescript
// Get keyword extractions for an RSS item
const { data: extractions } = await supabase
  .from('keyword_extractions')
  .select('*')
  .eq('source_type', 'rss_item')
```

## 4. Repurposed Content

### Get Repurposed Social Media Content

```typescript
import { createClient } from '@/lib/supabase/client';

const supabase = createClient();

// Get all repurposed content for an article
const { data: repurposedContent } = await supabase
  .from('repurposed_content')
  .select(`
    *,
    template:repurposing_templates (
      name,
      platform,
      template_structure
    )
  `)
  .eq('source_article_id', articleId)
  .order('created_at', { ascending: false });

  .from('repurposed_content')
  .select('*')
  .eq('source_article_id', articleId)

```

## 7. Direct Database Queries

### SQL Queries (Supabase SQL Editor)

#### Get All RSS Items with Generated Articles

```sql
SELECT 
  rfi.id,
  rfi.title,
  rfi.original_url,
  rfi.processing_status,
  rfi.generated_article_id,
  a.title as article_title,
  a.slug as article_slug,
  a.status as article_status,
  rf.name as feed_name
FROM rss_feed_items rfi
LEFT JOIN articles a ON a.id = rfi.generated_article_id
LEFT JOIN rss_feeds rf ON rf.id = rfi.feed_id
WHERE rfi.processing_status = 'article_generated'
ORDER BY rfi.imported_at DESC;
```

#### Get Pending RSS Items Ready for Article Generation

```sql
SELECT 
  rfi.*,
  rf.name as feed_name,
  rf.auto_generate_articles
FROM rss_feed_items rfi
JOIN rss_feeds rf ON rf.id = rfi.feed_id
WHERE rfi.processing_status = 'pending'
  AND rf.auto_generate_articles = true
ORDER BY rfi.published_date DESC;
```

#### Get All Automation Content Summary

```sql
-- Summary of automation content
SELECT 
  'RSS Feeds' as content_type,
  COUNT(*) as total_count,
  COUNT(CASE WHEN status = 'active' THEN 1 END) as active_count
FROM rss_feeds

UNION ALL

SELECT 
  'RSS Items' as content_type,
  COUNT(*) as total_count,
  COUNT(CASE WHEN processing_status = 'pending' THEN 1 END) as pending_count
FROM rss_feed_items

UNION ALL

SELECT 
  'Generated Articles' as content_type,
  COUNT(*) as total_count,
  COUNT(CASE WHEN status = 'published' THEN 1 END) as published_count
FROM articles
WHERE ai_generated = true

UNION ALL

SELECT 
  'Keyword Research' as content_type,
  COUNT(*) as total_count,
  COUNT(DISTINCT article_id) as articles_with_keywords
FROM keyword_research

UNION ALL

SELECT 
  'Generated Images' as content_type,
  COUNT(*) as total_count,
  COUNT(CASE WHEN is_active = true THEN 1 END) as active_count
FROM generated_images;
```

#### Get Articles with Full Automation Data

```sql
SELECT 
  a.id,
  a.title,
  a.slug,
  a.status,
  a.ai_generated,
  a.created_at,
  -- RSS source
  rfi.title as rss_source_title,
  rfi.original_url as rss_source_url,
  rf.name as rss_feed_name,
  -- Keywords
  (SELECT COUNT(*) FROM keyword_research WHERE article_id = a.id) as keyword_count,
  -- Images
  (SELECT COUNT(*) FROM generated_images WHERE article_id = a.id AND is_active = true) as image_count,
  -- Repurposed content
  (SELECT COUNT(*) FROM repurposed_content WHERE source_article_id = a.id) as repurposed_count
FROM articles a
LEFT JOIN rss_feed_items rfi ON rfi.generated_article_id = a.id
LEFT JOIN rss_feeds rf ON rf.id = rfi.feed_id
WHERE a.ai_generated = true
ORDER BY a.created_at DESC;
```

## 🎯 Quick Reference: Common Use Cases

### 1. Get All Pending RSS Items

```typescript
const { data } = await supabase
  .from('rss_feed_items')
  .select('*')
  .eq('processing_status', 'pending')
  .order('imported_at', { ascending: false });
```

### 2. Get Articles Generated Today

```typescript
const today = new Date().toISOString().split('T')[0];
const { data } = await supabase
  .from('articles')
  .select('*')
  .eq('ai_generated', true)
  .gte('created_at', today)
  .order('created_at', { ascending: false });
```

### 3. Get All Automation Content for Dashboard

```typescript
// Get multiple automation data sources
const [feeds, items, articles, runs] = await Promise.all([
  supabase.from('rss_feeds').select('count').eq('status', 'active'),
  supabase.from('rss_feed_items').select('count').eq('processing_status', 'pending'),
  supabase.from('articles').select('count').eq('ai_generated', true).eq('status', 'draft'),
  supabase.from('pipeline_runs').select('count').eq('status', 'running')
]);
```

### 4. Export Automation Content to JSON

```typescript
// Get all automation content for export
const exportData = {
  feeds: await rssImportService.getFeeds(),
  items: await supabase.from('rss_feed_items').select('*'),
  articles: await supabase.from('articles').select('*').eq('ai_generated', true),
  keywords: await supabase.from('keyword_research').select('*'),
  images: await supabase.from('generated_images').select('*').eq('is_active', true)
};

// Save to file or return as JSON
return JSON.stringify(exportData, null, 2);
```

## 📝 Notes

1. **Authentication**: Most queries require admin/editor authentication. Make sure you're using the server-side Supabase client (`createClient` from `@/lib/supabase/server`) for API routes.

2. **RLS Policies**: Row Level Security policies are enabled on all automation tables. Ensure your user has the appropriate permissions.

3. **Performance**: For large datasets, use pagination:
   ```typescript
   const { data } = await supabase
     .from('rss_feed_items')
     .select('*')
     .range(0, 49); // First 50 items
   ```

4. **Error Handling**: Always wrap database queries in try-catch blocks:
   ```typescript
   try {
     const { data, error } = await supabase.from('table').select('*');
     if (error) throw error;
     return data;
   } catch (error) {
     console.error('Error:', error);
     return [];
   }
   ```
